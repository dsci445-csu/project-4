df_TDS <- TDS_unfilt %>%
  rename(date = 1, value = 2) %>%
  mutate(date = as.Date(date, format = "%d-%b-%y"))

ggplot(df_TDS, aes(x = date, y = value)) +
  geom_line() +
  labs(x = "Date", y = "TDS Value", title = "TDS over Time")

df_TOC <- TOC_unfilt %>%
